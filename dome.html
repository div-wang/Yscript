Y('.selector_class') //寻找class;
Y('selector li') //寻找节点
Y('selector li').eq(0) //寻找first li
Y('selector li').eq(4) //寻找last li

//用法与jquery保持一致
Y('.anim_left').click(function(){
    Y('.box2').animate({
        speed : 20,  //缓动速度，数字越大速度越慢
        alter : 0,   //改变left值，每次加50
        target : 1000 //固定left值
    });
});
