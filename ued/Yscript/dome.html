    Y('.box2').animate({
        speed : 20,  //缓动速度，数字越大速度越慢
        alter : 0,   //改变left值，每次加50
        target : 1000 //固定left值
    });
});

Y('.dochtml').click(function(){
    var text = Y('.box6').html();
    Y('.box6 span').text('这是html:'+text);
}); 
Y('.doctext').click(function(){
    var text = Y('.box6').text();
    Y('.box6 span').text('这是text:'+text);
